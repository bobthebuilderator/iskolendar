    #User-Defined Functions used in the Program
    #User-Defined Log-in Function
    def login():

        log_in_option = input("""
    ===== Log-In =====
    Enter the number of your chosen option:
    1. Teacher
    2. Student
    Or input anything else to exit.
    Option: """)
        #Teacher Log-in
        if log_in_option == "1":
            #User-input
            username = input("Username: ")
            password = input("Password: ")
            #Condition for Successful Log-in
            if username in teacher_accounts and teacher_accounts[username] == password:
                print("\nLogin successful\n")
                teacher_menu()
                return True
            #Unsuccessful Log-in
            else:
                print("Invalid login\n")
                return False
        #Student Log-in
        elif log_in_option == "2":
            #User-input
            username = input("Username: ")
            password = input("Password: ")
            #Condition for Successful Log-in
            if username in student_accounts and student_accounts[username] == password:

                student_menu()

            #Unsuccessful Log-in
            else:

                return False

    #User-Defined Add Task Function
    def add_task(storage):
        #User-input
        name = input("Task name: ")
        task_type = input("Type (AA/FA/Review): ")
        priority = input("Priority (1-3, 3 highest): ")
        deadline = input("Deadline (YYYY-MM-DD): ")
        #Stores the task information
        storage.append({
            "name": name,
            "type": task_type,
            "priority": priority,
            "deadline": deadline
        })
        #User-confirmation
        print("Task added\n")
    #User-Defined Add Event Function
    def add_event(storage):
        #User-input
        name = input("Event name: ")
        place = input("Place: ")
        time = input("Time: ")
        requirement = input("Requirement: ")
        #Stores the event information
        storage.append({
            "name": name,
            "place": place,
            "time": time,
            "requirement": requirement
        })
        #User-confirmation
        print("Event added\n")
    #User-Defined Add Schedule Function
    def add_schedule(storage):
        #User-input
        name = input("Schedule name: ")
        events = input("Events (comma separated): ")
        time = input("Time/Date: ")
        #Stores the schedule information
        storage.append({
            "name": name,
            "events": events,
            "time": time
        })
        #User-confirmation
        print("Schedule added\n")
    #User-Defined Function for Sorting Tasks
    def sort_tasks(tasks):
        #If there are no tasks
        if len(tasks) == 0:
            return tasks
        #Sort Menu
        choice = input("""
    ===== SORT TASKS =====
    1. Sort by Priority
    2. Sort by Type
    3. Sort by Deadline
    Or input anything else to skip sorting.
    Option: """)
        #Sort by Priority (3 highest to 1 lowest)
        if choice == "1":
            priority_order = {
                "3": 1,
                "2": 2,
                "1": 3
            }
            sorted_tasks = sorted(
                tasks,
                key=lambda t: priority_order.get(t["priority"], 4)
            )
            return sorted_tasks
        #Sort by Type (AA, FA, Review)
        elif choice == "2":
            type_order = {
                "AA": 1,
                "FA": 2,
                "Review": 3
            }
            sorted_tasks = sorted(
                tasks,
                key=lambda t: type_order.get(t["type"], 4)
            )
            return sorted_tasks
        #Sort by Deadline (Nearest to Farthest)
        elif choice == "3":
            sorted_tasks = sorted(
                tasks,
                key=lambda t: t["deadline"]
            )
            return sorted_tasks
        #No Sorting Chosen
        else:
            return tasks
    #User-Defined View Calendar Function
    def view_calendar(tasks, events, schedules):
        print("\n===== ISKOLENDAR =====")
        #Sorts Tasks Before Displaying
        tasks = sort_tasks(tasks)
        #Displays Tasks
        print("\n--- TASKS ---")
        for t in tasks:
            print(t)
        #Displays Events
        print("\n--- EVENTS ---")
        for e in events:
            print(e)
        #Displays Schedules
        print("\n--- SCHEDULES ---")
        for s in schedules:
            print(s)
        print()
    #User-Defined Function for Managing Students
    def manage_students():
        #Loop for Student Management Menu
        while True:
            choice = input("""
    ===== STUDENTS =====
    1. View Students
    2. Add Student
    3. Remove Student
    4. Update Status
    Or input anything else to exit.
    Option: """)
            #View Students
            if choice == "1":
                for s, status in students.items():
                    print(s, "-", status)
                print()
            #Add Student
            elif choice == "2":
                name = input("Student name: ")
                students[name] = "Incomplete"
                print("Added\n")
            #Remove Student
            elif choice == "3":
                name = input("Student to remove: ")
                students.pop(name, None)
                print("Removed\n")
            #Update Status
            elif choice == "4":
                name = input("Student name: ")
                status = input("Status (Complete/Incomplete): ")
                if name in students:
                    students[name] = status
                    print("Updated\n")
                else:
                    print("Student not found\n")
            #Exit Menu
            else:
                break
